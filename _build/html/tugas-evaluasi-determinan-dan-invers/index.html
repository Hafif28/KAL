import numpy as np

print("\n" + "=" * 70)
print("PENYELESAIAN SOAL B: INVERS MATRIKS (METODE ADJOIN)")
print("=" * 70)

# ============================================================
# SOAL B.4: Invers Matriks 2×2
# ============================================================
print("\n>>> SOAL B.4: Invers Matriks 2×2")
print("A = [-7  -5]")
print("    [ 1   4]")

A4 = np.array([[-7, -5],
               [1, 4]], dtype=float)

det_A4 = np.linalg.det(A4)
print(f"\n📝 PERHITUNGAN MANUAL:")
print(f"det(A) = -23 (dari soal A.1)")

print("\nRumus untuk matriks 2×2:")
print("Jika A = [a  b], maka adj(A) = [ d  -b]")
print("         [c  d]                 [-c   a]")
print("\ndan A⁻¹ = (1/det A) · adj(A)")

print("\nUntuk matriks kami:")
print("adj(A) = [ 4   5]")
print("         [-1  -7]")

print(f"\nA⁻¹ = (1/{det_A4:.0f}) · [ 4   5]")
print("           [-1  -7]")

A4_inv = np.linalg.inv(A4)
print(f"\nA⁻¹ = {A4_inv[0,0]:7.4f}  {A4_inv[0,1]:7.4f}")
print(f"      {A4_inv[1,0]:7.4f}  {A4_inv[1,1]:7.4f}")

# Verifikasi
print("\n✓ VERIFIKASI: A · A⁻¹ = I")
I_check = A4 @ A4_inv
print(f"{I_check[0,0]:.0f}  {I_check[0,1]:.0f}")
print(f"{I_check[1,0]:.0f}  {I_check[1,1]:.0f}")

# ============================================================

A5 = np.array([[0, 2, -3],

print("\nLangkah 1: Hitung Matriks Kofaktor")

print("                   [ 0   1]")
print("C₁₂ = (-1)¹⁺² · det[ 1  -1] = -[(1)(1) - (-1)(0)] = -1")

# Verifikasi

I_check5 = A5 @ A5_inv
for i in range(3):

    print("Karena det(A) = 0, matriks ini TIDAK MEMILIKI INVERS.")

    A6_inv = np.linalg.inv(A6)
    print(f"\nMatriks Invers A⁻¹:")

    print(f"{A6_inv[1,0]:7.4f} {A6_inv[1,1]:7.4f} {A6_inv[1,2]:7.4f} {A6_inv[1,3]:7.4f}")

    # Verifikasi
    print("\n✓ VERIFIKASI: A · A⁻¹ ≈ I (dengan pembulatan)")
    I_check6 = A6 @ A6_inv
    for i in range(4):
        print(f"{I_check6[i,0]:7.0f} {I_check6[i,1]:7.0f} {I_check6[i,2]:7.0f} {I_check6[i,3]:7.0f}")

print("\n" + "=" * 70)
print("✓ SEMUA SOAL SELESAI")
print("=" * 70)
